In rasp : 

gst-launch-1.0 -v \
v4l2src device=/dev/video0 io-mode=2 ! \
image/jpeg,width=1280,height=720,framerate=30/1 ! \
jpegdec ! \
videoconvert ! \
x264enc tune=zerolatency speed-preset=ultrafast bitrate=3000 key-int-max=30 ! \
h264parse ! \
rtph264pay config-interval=1 pt=96 ! \
udpsink host=100.115.66.25 port=5600 sync=false async=false

In laptop : 

gst-launch-1.0 -v \
udpsrc port=5600 caps="application/x-rtp,media=video,encoding-name=H264,payload=96" ! \
rtph264depay ! \
h264parse ! \
avdec_h264 ! \
videoconvert ! \
autovideosink sync=false
